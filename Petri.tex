    \frametitle{Общая терминология}
    \begin{itemize}
        \item Переходы могут быть связаны только с позициями (и наоборот)
        \item Переход может \textbf{сработать} когда в каждой входной позиции есть по меньшей мере одна метка
        \item \textbf{Срабатывание} перехода забирает по одной метке из каждой
        входной позициии добавляет по одной метке в каждую выходную позицию
    \end{itemize}
  \end{frame}
  \begin{frame}[fragile]  
  \frametitle{Способы задания сетей Петри}
  \begin{itemize}
      \item Аналитический \\ \tab Перечисление элементов множеств позиций, переходов, перечисление элементов комплектов входной и выходной функций
      \item Графический \\ \tab Представляет собой двудольный, ориентированный мультиграф
      \item Матричный \\ \tab Матрица имеет n столбцов (по числу вершин мест) и k строк (по числу вершин переходов). Элементами являются нули и единицы. Единица - если имеется дуга от вершины места к вершине перехода, иначе ноль
  \end{itemize}
\end{frame}

    \textbf{Сеть Петри} - это двудольный ориентированный мультиграф A=(P,T,I,O), где:
    \begin{itemize}
        \item \textbf{P} – множество состояний;
        \item \textbf{T} – множество переходов;
        \item \textbf{I} – множество (картеж) входных дуг;
        \item \textbf{O} – множество (картеж) выходных дуг;  
      \end{itemize}
      \textbf{Маркировка СП} – расстановка фишек в состояниях СП. Это вектор неотрицательных целых чисел, размерность которого равна числу состояний СП.
\end{frame}
  \begin{frame}[fragile]  
  \frametitle{Более строгое определение}
      \begin{figure}[ht]
        \centering
      \includegraphics[width=0.53\linewidth]{pp.png} 
    \end{figure}
    \begin{itemize}
        \item Состояния (P): P=\{$P_1,P_2,P_3,P_4$\}
        \item Переходы (T): T=\{$t_1,t_2$\}
        \item Маркировка ($\mu$): $\mu$=(1,0,2,1)$^T$
        \item Входы (I): I($t_1$)=\{$P_1$\}; I($t_2$)=\{$P_3,P_2$\}
        \item Выходы (O): O($t_1$)=\{$P_2,P_3$\}; O($t_2$)=\{$P_1,P_4$\}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]  
  \frametitle{Проверка критериев сети Петри при их моделирование}
  \begin{itemize}
      \item \textbf{Ограниченность} --- число меток в любой позиции сети не может превысить некоторого значения K
      \item \textbf{Безопасность} --- СП называется безопасной, если каждая позиция содержит не более одной метки
      \item \textbf{Консервативность} --- СП называется консервативной, если общее количество меток во всех позициях всегда постоянно
      \item \textbf{Достижимость} --- возможность перехода сети из одного заданного состояния (характеризуемого распределением меток) в другое
      \item \textbf{Живучесть} --- возможность срабатывания любого перехода при функционировании моделируемого объекта  
  \end{itemize}
\end{frame}
\begin{frame}[fragile]  
  \frametitle{Некоторые виды сетей Петри}
  \begin{itemize}
      \item \textbf{Временная} сеть Петри --- переходы обладают весом, определяющим продолжительность срабатывания (задержку)
      \item \textbf{Стохастическая} сеть Петри --- задержки являются случайными величинами
      \item \textbf{Цветная} сеть Петри --- метки могут быть различных типов, обозначаемых цветами, тип метки может быть использован как аргумент в функциональных сетях
      \item \textbf{Ингибиторная} сеть Петри --- возможны ингибиторные дуги, запрещающие срабатывания перехода, если во входной позиции, связанной с переходом ингибиторной дугой, находится метка
      \item \textbf{Иерархическая} сеть --- содержит немгновенные переходы, в которые вложены другие, возможно, также иерархические, сети
  \end{itemize}
\end{frame}

  Взаимодействие процессов требует распределения ресурсов между ними. Чтобы система работала правильно распределением ресурсов необходимо управлять. Это управление в основном сводится к задачам синхронизации взаимодействия процессов

  \begin{frame}[fragile]  
  \frametitle{Задача о производителе/потребителе}
  Процесс производитель порождает компоненту информации и размещает ее в буфер. Процесс потребитель ждет пока компонента информации разместится в буфере, после этого он может ее удалить из буфера и использовать \\
  Как правило используется буфер ограниченной емкости, то есть имеется возможность разместить в нем не более n компонентов данных
\end{frame}

\begin{frame}[fragile]  
  \frametitle{Обход сети}
  Существуют разные алгоритмы обхода, рассмотрим один из них:
  \begin{enumerate}
      \item Даем каждому переходу одинаковую вероятность (4 перехода = 0,25 каждому)
      \item Генерируем случайное число - порог вероятности выбора перехода
      \item Выбираем первый переход с подходящей под порог вероятностью, который может сработать
      \item Срабатываем переход
      \item Проверяем пустые и заполненные позиции
      \item Уменьшаем вероятность сработанного перехода в 2 раза
      \item Если все переходы сработали хотя бы раз, останавливаемся. Если нет, повторяем спункта 2
  \end{enumerate}
\end{frame}
